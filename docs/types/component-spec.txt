////// Component type


// Properties
name: 'string'             // unique name (e.g. 'leftFoot', 'centerSpeaker'). Needed to reference to the component (using @instance#component syntax)
alternativeName: 'string*' // other possible names for the components
status: ''                 // 'always' (100%), 'typically' (>95%), 'possible' (>0%), 'never' (for updates or incomplete)
ref: 'ref'                 // what is the component? (ref to instance or category, or inline object)
quantity: 'number'         // optional. How many? Default: 1
position: 'positionType*'   // optional. Where is its position.
connections: 'connectionType*' // relationships to other components in the same thing
properties: {}                 // if reference is a category, these properties can give details (e.g. if category 'cat', it may contain the color). 


/////// Connection type
to: 'string'       // reference to the other component's unqiue name
type: 'string'     // 'adjacent', 'joined', 'unibody', 'in', 'virtual' (for non-physical things)
